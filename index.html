async function signUp() { const fullname = document.getElementById('fullname').value; const email = document.getElementById('email').value; const password = document.getElementById('password').value; // 1. تسجيل الحساب const { data, error } = await supabase.auth.signUp({ email, password }); if (error) { alert("خطأ في المصادقة: " + error.message); return; } // 2. محاولة الحفظ في الجدول مع طباعة الخطأ إذا فشل const { error: profileError } = await supabase .from('profiles') .insert([{ id: data.user.id, full_name: fullname }]); if (profileError) { console.error("تفاصيل خطأ قاعدة البيانات:", profileError); alert("فشل حفظ البيانات: " + profileError.message); } else { alert("تم التسجيل بنجاح!"); } }
